def make_grid(X,y,res):
    y_min = y.min()-0.2
    y_max = y.max()+0.2
    x_min = X.min()-0.2
    x_max = X.max()+0.2
    x_arr = np.linspace(x_min,x_max,res)
    y_arr = np.linspace(y_min,y_max,res)
    xx,yy = np.meshgrid(x_arr,y_arr)  
    return xx,yy
